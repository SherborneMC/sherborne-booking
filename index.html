<div id="panel" class="panel">

  <button onclick="closePanel()">X</button>

</div>


<div id="meta" class="meta"></div>


<script>

let cells=[]

let requested=new Set()

let internal=false

let timer=null


async function load(){

  const r=await fetch('/api/availability')

  const j=await r.json()

  cells=j.cells || []

  render()

  updateMeta()

}


function render(){

  app.innerHTML=cells.filter(c=>c.bookable).map(c=>{

    const used=requested.has(c.id)

    return `<div class="slot ${used?'requested':''}" data-id="${c.id}">

      ${used?'Requested':'Request'} 

      ${new Date(c.startUtc).toLocaleTimeString([], {hour:'2-digit',minute:'2-digit'})}

    </div>`

  }).join('')


  let clicks=0


  document.querySelectorAll('.slot').forEach(el=>{

    el.onclick=()=>{

      clicks++

      setTimeout(()=>clicks=0,250)


      // internal mode

      if(internal && clicks===2){

        createReminder(el.dataset.id)

        return

      }


      if(internal){

        deactivate()

      }


      // open panel

      if(clicks===2){

        openPanel()

      }

    }

  })

}


function openPanel(){

  panel.classList.add('open')

}


function closePanel(){

  panel.classList.remove('open')

}


document.addEventListener('keydown', e=>{

  if(e.key==='Escape') closePanel()

})


// INTERNAL MODE

trigger.ondblclick=()=>{

  internal=true

  meta.classList.add('hidden')

  clearTimeout(timer)

  timer=setTimeout(deactivate,15000)

}


function deactivate(){

  internal=false

  meta.classList.remove('hidden')

}


async function createReminder(id){

  await fetch('/api/book',{

    method:'POST',

    headers:{'Content-Type':'application/json'},

    body:JSON.stringify({slotId:id,internal:true})

  })

  requested.add(id)

  deactivate()

  render()

}


function updateMeta(){

  meta.textContent='v11 · '+new Date().toLocaleTimeString([], {hour:'2-digit',minute:'2-digit'})

}


// refresh

setInterval(()=>{

  if(document.visibilityState==='visible'){

    meta.textContent='Updating…'

    load()

    setTimeout(updateMeta,5000)

  }

},120000)


load()

</script>
